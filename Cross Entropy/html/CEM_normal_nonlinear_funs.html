def calculate_KL(points,omegas,s,M,D,log_NK,size_Pi,tol_KL = 1/10**8):
    KL = 0
    for m in range(M):
        for d in range(D):
            size_Pi_m_d = (s[m][d+1] - s[m][d])* (omegas[m+1] - omegas[m])
            fun = lambda x:  ( s[m][d+1] >= x[0] >= s[m][d])  & ( omegas[m+1] >= x[1] >= omegas[m])   
            size_P_m_d = sum(np.array(list(map(fun, points)))) ## to avoid INF
            if size_P_m_d == 0:
                size_P_m_d = tol_KL
            KL_m_d = size_Pi_m_d * np.log(size_Pi_m_d) - size_Pi_m_d * np.log(size_P_m_d)
            KL += KL_m_d
    KL /= size_Pi  
    KL += log_NK - np.log(size_Pi)  
    return KL

def calculate_KL_kde(points,omegas,s,kde,M,D,log_pdf_Pi,log_size_Pi):
    omegas = np.sort(omegas)
    KL = 0
    for m in range(M):
        for d in range(D):
            s[m] = np.sort(s[m])
            size_Pi_m_d = (s[m][d+1] - s[m][d])* (omegas[m+1] - omegas[m])
            pdf_Pi_m_d = kde.integrate_box([s[m][d],omegas[m]],[s[m][d+1],omegas[m+1]])
            KL_m_d = size_Pi_m_d * np.log(size_Pi_m_d) - size_Pi_m_d * np.log(pdf_Pi_m_d)
            KL += KL_m_d
    KL /= size_Pi  
    KL += log_pdf_Pi - log_size_Pi  
    return KL

def calculate_KL_kde_samples(samples,points,kde,omega_0,omega_M,t_0,t_N,M,D,log_pdf_Pi,log_size_Pi):
    omegas = np.sort(samples[0])
    omegas = np.concatenate([[omega_0],omegas,[omega_M]])
    KL = 0
    for m in range(M):
        for d in range(D):
            s_m = np.sort(samples[1][m])
            s_m = np.concatenate([[t_0],s_m,[t_N]])            
            size_Pi_m_d = (s_m[d+1] - s_m[d]) * (omegas[m+1] - omegas[m])
            pdf_Pi_m_d = kde.integrate_box([s_m[d],omegas[m]],[s_m[d+1],omegas[m+1]])
            #print(pdf_Pi_m_d)
            KL_m_d = size_Pi_m_d * np.log(size_Pi_m_d) - size_Pi_m_d * np.log(pdf_Pi_m_d)
            KL += KL_m_d
    KL /= size_Pi  
    KL += log_pdf_Pi - log_size_Pi  
    return KL

def truncate_sample(p, p_low,p_high,delta_p):
    p_trun = max(min(p,p_high - delta_p),p_low + delta_p)
    return p_trun
vec_truncate_sample = np.vectorize(truncate_sample)

def generate_sample(mu_m,mu_m_d,sigma2_m,sigma2_m_d,M,D,a):
    w_sim = random_state.multivariate_normal(mu_m[1:M],sigma2_m * np.eye(M-1))
    w_sim_trun = vec_truncate_sample(w_sim,  mu_m[0],  mu_m[M] ,sigma2_m* a)
    s_sim_trun = list()
    for m in range(M):
        s_sim = random_state.multivariate_normal(mu_m_d[m][1:D],sigma2_m_d * np.eye(D-1))
        s_sim_trun.append(vec_truncate_sample(s_sim,  mu_m_d[m][0],  mu_m_d[m][D] ,sigma2_m_d*a))        
    return [w_sim_trun,s_sim_trun]

def generate_samples(n_sim,mu_m,mu_m_d,sigma2_m,sigma2_m_d,M,D,a):
    return [generate_sample(mu_m,mu_m_d,sigma2_m,sigma2_m_d,M,D,a) for _ in range(n_sim)]
